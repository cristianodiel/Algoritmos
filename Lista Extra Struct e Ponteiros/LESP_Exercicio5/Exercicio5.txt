instruções (assuma que o endereço de x é 1000 e de y é 1004):
int x = 10, y=20;
int* p1;
int* p2;
p1 = &x;
p2 = &y;
(*p1)++;

(a) x
    11

(b) y
    20

(c) &x
    1000

(d) &y
    1004

(e) p1
    1000

(f) p2
    1004

(g) *p1 + *p2
    31

(h) *(&x)
    11

(i) &(*p2)
